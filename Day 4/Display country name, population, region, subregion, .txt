Task : Use the rest countries and print all countries name, region, sub region and population

Code:

// To print all the country names
fetch("https://restcountries.com/v2/all")
    .then(function (res) {
        return res.json()
    })
    .then(function (data) {
        for (element in data) {
            console.log(data[element].name)
        }
    })

// To print all the country region
fetch("https://restcountries.com/v2/all")
    .then(function (res) {
        return res.json()
    })
    .then(function (data) {
        for (element in data) {
            console.log(data[element].region)
        }
    })

// To print all the country sub-region:
fetch("https://restcountries.com/v2/all")
    .then(function (res) {
        return res.json()
    })
    .then(function (data) {
        for (element in data) {
            console.log(data[element].subregion)
        }
    })

// To print all the country population
fetch("https://restcountries.com/v2/all")
    .then(function (res) {
        return res.json()
    })
    .then(function (data) {
        for (element in data) {
            console.log(data[element].population)
        }
    })


Description: In the above code we have used the fetch method to fetch all the data response from the url. the function(res) converts all the data in JSON format. The function data will iterate over all the properties of the data and print the values present of the provided argument.
